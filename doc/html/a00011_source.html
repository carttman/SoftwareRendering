63 #define N norm
64 #define L g_LightDir //조명 방향 (월드)
65
66 L = mul ( L, g_mView); //조명 벡터, 뷰 공간으로... L = L * mView
67 L = normalize( -L ); //내적을 위해 방향 전환.★
68
69 N = mul ( N, g_mWV ); //노멀, 뷰 공간으로... N = N * mWorld * mView
70 N = normalize( N );
71
72 float Ldot = max( dot( N, L), 0 ); //조명량 계산 : 내적, 0 이하 버리기.
73
74 //조명계산.★
75 float4 Diff = g_LightDiffuse * g_MtrlDiffuse * Ldot;
76 float4 Amb = g_AmbientLightColor * g_MtrlAmbient;
77 float4 color = Diff + Amb;
78
79
80 //----------------------------------
81 // 출력..★
82 //----------------------------------
83 temp.diff = color;
84
85 return temp;
86}
