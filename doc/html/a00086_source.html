88//! <DX> 주어진 옵션에 맞추어 VRAM 을 확보합니다.
89//! <Yena> 일반 시스템 메모리를 생성합니다.
90//!
91//! \param desc 생성할 정점버퍼 정보기술 구조체
92//! \return 성공시 OK, 실패시 FALSE
93//
94int B3YenaVertexBuffer9::_Create( B3YVERTEXBUFFER_DESC desc )
95{
96 //이미 만들어져 있다면 오류..사용자에게 해결을 요구.
97 //예외처리 추가를 추천합니다. ex) YN_ERR_NOTEMPTY
98 if (YN_VALIED(m_pVBuffer)) return YN_FAIL;
99
100 //정점 버퍼 메모리 확보.
101 //DX 처럼 VRAM 을 확보할 수 없으므로, Pool 옵션은 무시합니다.
102 m_pVBuffer = static_cast<void*>(new BYTE[desc.Size]); assert(m_pVBuffer);
103 m_SizeInByte = desc.Size;
104 m_FVF = desc.FVF;
105 m_Desc = desc;
106
107 //정점 1개의 크기(바이트) 계산하기.
108 //지정된 정점규격(FVF) 기준 계산, 가능한 단순하게 처리합니다.
109 m_Stride = 0;
110 if (CHECK(m_FVF, B3YFVF_XY)) m_Stride += sizeof(float) * 2; //2d 좌표.(앞으로 사용하지 않겠지만, 사용 예로써 남겨둡니다)
111 if (CHECK(m_FVF, B3YFVF_XYZ)) m_Stride += sizeof(float) * 3; //3d 좌표.
112 if (CHECK(m_FVF, B3YFVF_DIFFUSE)) m_Stride += sizeof(DWORD); //DWORD 형 색상.
113
114 //오류 체크..
115 if (m_Stride <= 0) { /* Error!!... */ return YN_FAIL; }
116
117 //정점 개수 확인.(디버깅용)
118 m_VtxCnt = _GetVtxCnt();
119
120 return YN_OK;
121}

126//
127//! 사용자가 지정한 옵션으로 버퍼를 생성합니다.
128//! <DX> 주어진 옵션에 맞추어 VRAM 을 확보합니다.
129//! <Yena> 일반 시스템 메모리를 생성합니다.
130//!
131//! \param Length 정점 버퍼 전체 크기 (바이트)
132//! \param FVF 정점 규격.
133//! \param Pool 메모리 풀(사용) 옵션.
134//! \return 성공시 OK, 실패시 FALSE
135//
136int B3YenaVertexBuffer9::_Create( UINT Length, DWORD FVF, B3YPOOL Pool )
137{
138 // 생성할 정점 버퍼 정보 기술
139 // <Yena> 시스템 메모리를 사용하므로 유의미한 동작은 없으나, 구조의 일관성을 위해 사용.
140 // GPU 메모리(VRAM) 확보를 위해서 이 정도의 수고는 필요하다...정도로 이해해 봅시다.
141 B3YVERTEXBUFFER_DESC desc = {};
142 desc.Format = B3YFMT_VERTEXDATA;
143 desc.Type = B3YRTYPE_VERTEXBUFFER;
144 desc.Usage = B3YUSAGE_WRITEONLY; //정점버퍼는 CPU "쓰기전용", GPU "읽기 전용" 임을 상기합시다.
145 desc.Pool = Pool;
146 desc.Size = Length;
147 desc.FVF = FVF;
148 int res = _Create(desc);
149
150 return res;
151}

279////////////////////////////////////////////////////////////////////////////////
280//
281//! 정점 버퍼 잠그기.
282//!
283//! <DX> 정점 버퍼(VRAM) 에 데이터를 입력하기 위해서는 CPU 가 접근 가능해야 합니다.
284//! Lock 함수는 버퍼를 CPU 의 "쓰기 또는 읽기" 가능 상태로 변경합니다.
285//! 전체 버퍼 및 부분 잠금이 가능하며 잠금된 영역은 GPU 의 접근이 제한됩니다.("접근 잠금")
286//! 사용이 끝난 버퍼는 UnLock 함수로 잠금을 해제해야 합니다.
287//!
288//! <Yena> "잠금" 및 "해제" 동작을 SW 로 구현,묘사 가능하겠으나
289//! 시스템 메모리를 사용중으로 특별한 기능은 추가하지 않겠습니다.
290//! DX 의 실제 역활과는 다르지만, 동작의 흐름은 유지합니다.
291//!
292//! \see [참고] DX9::Lock (https://learn.microsoft.com/en-us/windows/win32/api/d3d9/nf-d3d9-idirect3dvertexbuffer9-lock)
293//!
294//! \param OffsetToLock '잠금' 할 버퍼 (옵셋) 주소. 기본값 0,
295//! \param SizeToLock '잠금' 할 버퍼 크기
296//! \param ppbData 리턴받을 버퍼 포인터
297//! \param Flags '잠금' 옵션. 기본값 0.
298//! \return 성공시 OK, 실패시 FALSE
299//
300int B3YenaVertexBuffer9::Lock( UINT OffsetToLock, UINT SizeToLock, void** ppbData, DWORD Flags )
301{
302 if (YN_ENABLED(m_bLocked)) //잠금 상태 확인...
303 {
304 ynLog(_T("이 버퍼는 이미 잠겨(Locked) 있습니다. %s=%x)"), ynToString(m_pVBuffer), m_pVBuffer);
305 //return YN_FAIL; //지속적인 경고 출력을 위해 리턴하지 않음.
306 }
307
308 if (YN_VALIED(*ppbData)) return YN_FAIL; //유효성 검사@fix.2015.11.
309
310 BYTE* pVB = (BYTE*)m_pVBuffer + OffsetToLock; //지정 주소 계산.
311 *ppbData = (void*)pVB; //주소 넘김.
312
313 //잠금 상태 설정.
314 m_bLocked = TRUE;
315
316 return YN_OK;
317}

336int B3YenaVertexBuffer9::Unlock(void)
337{

339 {
340 ynLog(_T("이 버퍼는 이미 잠금 해제(Unlocked) 상태 입니다. %s=%x)"), ynToString(m_pVBuffer), m_pVBuffer);

343
344 //<Yena> "잠금" 및 "해제" 동작을 SW 로 구현,묘사 가능하겠으나
