154//
155//! 엔진 타이머 : 초당 프레임(fps) 출력
156//!
157//! \param x, y 출력할 화면좌표(2D)
158//! \return 없음.
159//
160void PutFPS(int x, int y)
161{
162 static UINT frm = 0;
163 static float fps = 0.0f;
164 ++frm;
165 static ULONGLONG oldtime = GetTickCount64();
166 ULONGLONG nowtime = GetTickCount64();
167
168 UINT time = (UINT)(nowtime - oldtime);
169 if (time >= 1000)
170 {
171 fps = (float)(frm * 1000) / (float)time;
172 frm = 0;
173 oldtime = nowtime;
174 }
175
176 DrawText(x, y, RGB(0, 255, 0), _T("FPS=%.1f/%d"), fps, time);
177}
