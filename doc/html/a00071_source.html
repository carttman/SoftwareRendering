66int ynAxis::Init(LPDEVICE pDev, AXISINFO* pAi)
67{
68 assert(pDev != NULL);
69 assert(pAi != NULL);
70
71 //디바이스 저장.
72 m_pDev = pDev;
73
74
75 //방향축 색상 (DX 형식 0xAARRGGBB)
76 //우리가 만든 DWORD 형변환 연산자는 DWORD(dx) --> B3YXCOLOR 임을 기억할 것.
77 //COLOR col = COLOR(0, 0, 0, 1);
78 #define RED 0xffff0000 //COLOR(1, 0, 0, 1);
79 #define GREEN 0xff00ff00 //COLOR(0, 1, 0, 1);
80 #define BLUE 0xff0080ff //COLOR(0, 0.3f, 1, 1);
81
82 //정점개수. (직선x3 + 문자위치3)
83 int vtxcnt = 9;
84
85 //데이터 구성.
86 m_pVtxs = new VTX_AXIS[vtxcnt];
87 assert(m_pVtxs);
88 m_pVtxs[0].vPos = VECTOR3(0, 0, 0); //x 축. 라인.
89 m_pVtxs[1].vPos = VECTOR3(pAi->length, 0, 0);
90 m_pVtxs[0].Color = RED;
91 m_pVtxs[1].Color = RED;
92
93 m_pVtxs[2].vPos = VECTOR3(0, 0, 0); //y 축. 라인.
94 m_pVtxs[3].vPos = VECTOR3(0, pAi->length, 0);
95 m_pVtxs[2].Color = GREEN;
96 m_pVtxs[3].Color = GREEN;
97
98 m_pVtxs[4].vPos = VECTOR3(0, 0, 0); //z 축. 라인.
99 m_pVtxs[5].vPos = VECTOR3(0, 0, pAi->length);
100 m_pVtxs[4].Color = BLUE;
101 m_pVtxs[5].Color = BLUE;
102
103 m_pVtxs[6].vPos = VECTOR3(pAi->length+0.1f, 0, 0); //"X"
104 m_pVtxs[6].Color = RED;
105 m_pVtxs[7].vPos = VECTOR3(0, pAi->length+0.1f, 0); //"Y"
106 m_pVtxs[7].Color = GREEN;
107 m_pVtxs[8].vPos = VECTOR3(0, 0, pAi->length+0.1f); //"Z"
108 m_pVtxs[8].Color = BLUE;
109
110
111
112
113 //정점 버퍼 구성.
114 int res = YN_OK;
115 int vbsize = vtxcnt * sizeof(VTX_AXIS);
116 res = m_pDev->CreateVertexBuffer(vbsize, 0, FVF_AXIS, B3YPOOL_SYSTEMMEM, &m_pVB, 0);
117 if(YN_FAILED(res))
118 {
119 //B3D_Error(NO_, "B3Axis::Init() 실패. VB 생성 실패.");
120 return res;
121 }
122
123 VTX_AXIS* pVB = NULL;
124 res = m_pVB->Lock(0, vbsize, (VOID**)&pVB, 0);
125 if(YN_FAILED(res))
126 {
127 //.. 에러...
128 return res;
129 }
130 ::memcpy(pVB, m_pVtxs, vbsize);
131 m_pVB->Unlock();
132
133
134 //전체 라인개수.
135 //m_uLineCnt = vtxcnt / 2;
136
137 //뷰포트 구성..
138 m_ViewPort.X = 0;
139 m_ViewPort.Y = 0;
140 m_ViewPort.Width = (DWORD)(pAi->scnX * VIEWPORT_SCALE_);
141 m_ViewPort.Height = (DWORD)(pAi->scnY * VIEWPORT_SCALE_);
142 m_ViewPort.MinZ = 0.0f;
143 m_ViewPort.MaxZ = 1.0f;
144
145
146 //완료.
147 //B3D_Error(NO_, "B3Axis::Init() 성공. 공간축 구성 완료.");
148
149 return TRUE;
150}

156/////////////////////////////////////////////////////////////////////////////
157//
158//! 월드 좌표 공간에 정렬하여 축 선 보여주기.
159//
160int ynAxis::Update(
161 float dTime /*= 0*/,
162 VECTOR3 vPos /*=VECTOR3(0,0,0)*/, //출력좌표. (월드)
163 int width /*=0*/, int height /*=0*/
164 )
165{
166 //if(pDev != NULL)m_pDev = pDev;
167 //B3YXMatrixIdentity(&m_mTM);
168
169 //화면크기 산출 : 뷰포트 미적용시...사용.
170 if(width != 0)
171 {
172 m_ViewPort.X = 0;
173 m_ViewPort.Y = 0;
174 m_ViewPort.Width = width;
175 m_ViewPort.Height = height;
176 m_ViewPort.MinZ = 0.0f;
177 m_ViewPort.MaxZ = 1.0f;
178 }
179
180
181 //월드 변환 행렬.
182 MATRIX mTM; B3YXMatrixIdentity(&mTM);
183 mTM._41 = vPos.x; mTM._42 = vPos.y; mTM._43 = vPos.z;
184 m_mTM = mTM;
185
186
187 //축방향 문자 출력좌표 계산용 행렬 준비..
188 MATRIX mView; m_pDev->GetTransform(B3YTS_VIEW, &mView);
189 MATRIX mProj; m_pDev->GetTransform(B3YTS_PROJECTION, &mProj);
190 MATRIX mVP; B3YXMatrixIdentity(&mVP);
191 mVP._11 = m_ViewPort.Width * 0.5f;
192 mVP._22 = -(m_ViewPort.Height * 0.5f);
193 mVP._41 = m_ViewPort.X + m_ViewPort.Width * 0.5f;
194 mVP._42 = m_ViewPort.Y + m_ViewPort.Height * 0.5f;
195
196 //축방향 문자 출력좌표 계산용 행렬 누계..
197 mTM = mTM * mView * mProj * mVP;
198
199 //축방향 문자 출력좌표 최종 계산.
200 for(int i=6, k=0; i<9; i++ , k++)
201 B3YXVec3TransformCoord(&m_vTextPos[k], &m_pVtxs[i].vPos, &mTM);
202
203 return TRUE;
204}

213

217 B3DrawText((int)m_vTextPos[2].x, (int)m_vTextPos[2].y, COLOR(m_pVtxs[8].Color), _T("Z"));
218
219
220 //방향축 라인 그리기.

222 m_pDev->SetStreamSource(0, m_pVB, 0, sizeof(VTX_AXIS));
223 m_pDev->SetFVF(FVF_AXIS);
224 m_pDev->DrawPrimitive(B3YPT_LINELIST, 0, 3);
225 //m_pDev->DrawPrimitiveUP(B3YPT_LINELIST, 3, m_pVtxs, sizeof(VTX_AXIS));
226
227 return TRUE;
228}

235int ynAxisCreate(ynAxis** ppAxis, LPDEVICE pDev, int scnX, int scnY, float length /* = 1.0f */, int scale /* = 1.0f */)
236{
237 if(YN_VALIED(*ppAxis)) return YN_FAIL;
238 if(YN_INVALIED(pDev)) return YN_FAIL;
239
240 ynAxis::AXISINFO ai;
241 ai.length = 1.0f; //방향축 길이.
242 ai.scnX = scnX; //방향축 표시 화면 영역.(Width, Pixel)
243 ai.scnY = scnY; //방향축 표시 화면 영역.(Height, Pixel)
244
245 ynAxis* pAxis = new ynAxis;
246 assert(pAxis);
247 pAxis->Init(pDev, &ai);
248
249 *ppAxis = pAxis; //외부에 복사.
250
251 return YN_OK;
252}
