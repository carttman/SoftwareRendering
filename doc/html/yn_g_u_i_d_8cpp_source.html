74//! : IsEqualGUID, IsEqualIID 와 동일 기능
75//!
76//! \see [IsEqualGUID, IsEqualIID](https://learn.microsoft.com/en-us/windows/win32/api/guiddef/nf-guiddef-isequaliid)
77//!
78//! \param id1 검사할 IID
79//! \param id2 감사할 IID
80//! \return 동일하면 TRUE, 그렇지 않으면 FALSE.
81//
82BOOL ynIsEqualIID(const YNIID& id1, const YNIID& id2)
83{
84 return !memcmp(&id1, &id2, sizeof(YNIID));
85}
