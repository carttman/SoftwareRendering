58//! 그리드 객체 초기화
59//! 외부에서 지정된 정보를 기준, 격자와 기타 정보를 구성한다.
60//
61int ynGrid::Init(LPDEVICE pDev, GRIDINFO* pGrid)
62{
63 assert(pDev != NULL);
64 assert(pGrid != NULL);
65
66 //외부 지정 정보 저장.
67 m_Info = *pGrid;
68 m_pDev = pDev;
69
70
71 //정적 버퍼 생성.
72 int xcnt = (pGrid->width / (int)pGrid->scale) + 1; //가로/세로별 라인 개수.
73 int ycnt = (pGrid->width / (int)pGrid->scale) + 1;
74 int vtxcnt = (xcnt + ycnt) * 2; //정점개수.
75
76 m_pGrids = new VTX_GRID[vtxcnt];
77 assert(m_pGrids);
78 ZeroMemory(m_pGrids, sizeof(VTX_GRID) * vtxcnt);
79
80
81 //그리드 시작 위치. (기본. 원점(0,0,0))
82 float hx = (pGrid->width * 0.5f);
83 float hy = (pGrid->height * 0.5f);
84
85 //그리드 중앙색 (DX 형식 0xAARRGGBB)
86 //우리가 만든 DWORD 형변환 연산자는 DWORD(dx) --> B3YXCOLOR 임을 기억할 것.
87 //COLOR col = COLOR(0, 0, 0, 1);
88 DWORD col = 0xff000000;
89
90
91 //x 축 라인 생성.
92 int k=0;
93 for(int i=0; i<xcnt; i++, k+=2)
94 {
95 float sx = -hx;
96 float sz = hy - i * pGrid->scale; //위에서 아래로 내려옵니다.(+Z --> -Z)
97
98 m_pGrids[k].vPos = VECTOR3(sx, 0.0f, sz);
99 m_pGrids[k+1].vPos = VECTOR3(sx + pGrid->width, 0.0f, sz);
100
101 m_pGrids[k].Color = (xcnt/2 == i)?col:pGrid->color;
102 m_pGrids[k+1].Color = (xcnt/2 == i)?col:pGrid->color;
103 }
104
105 //z 축 라인 생성
106 for(int j=0; j<ycnt; j++, k+=2)
107 {
108 float sx = -hx + j * pGrid->scale; //왼쪽에서 오른쪽으로..(-X --> +X)
109 float sz = hy;
110
111 m_pGrids[k].vPos = VECTOR3(sx, 0.0f, sz);
112 m_pGrids[k+1].vPos = VECTOR3(sx, 0.0f, sz - pGrid->height);
113
114 m_pGrids[k].Color = (ycnt/2 == j)?col:pGrid->color;
115 m_pGrids[k+1].Color = (ycnt/2 == j)?col:pGrid->color;
116 }
117
118
119
120 //정점 버퍼 구성.
121 int res = YN_OK;
122 int vbsize = vtxcnt * sizeof(VTX_GRID);
123 res = m_pDev->CreateVertexBuffer(vbsize, 0, FVF_GRID, B3YPOOL_SYSTEMMEM, &m_pVB, NULL );
124 if (YN_FAILED(res))
125 {
126 //B3D_Error(NO_, "B3Grid::Init() 실패. VB 생성 실패.");
127 return res;
128 }
129
130 VTX_GRID* pVB = NULL;
131 res = m_pVB->Lock(0, vbsize, (void**)&pVB, 0);
132 if(YN_FAILED(res))
133 {
134 //.. 에러...
135 return res;
136 }
137 ::memcpy(pVB, m_pGrids, vbsize);
138 m_pVB->Unlock();
139
140
141 //전체 라인개수.
142 m_uLineCnt = vtxcnt / 2;
143
144
145 //완료.
146 //B3D_Error(NO_, "B3Grid::Init() 성공. 그리드 로딩 완료.");
147
148
149 return TRUE;
150}

178int ynGrid::Render(float dTime/*=0*/, float sx /* = 0.0f */, float sy /* = 0.0f */)
179{
180 m_pDev->SetTransform(B3YTS_WORLD, &m_mTM);
181 m_pDev->SetStreamSource(0, m_pVB, 0, sizeof(VTX_GRID));
182 m_pDev->SetFVF(FVF_GRID);

194int ynGridCreate(ynGrid** ppGrid, LPDEVICE pDev, int width /* = 100 */, int height /* = 100 */,
195 float scale /* = 10.0f */, COLOR col /* = COLOR */)
196
197{
198 if(YN_VALIED(*ppGrid)) return YN_FAIL;

200

203 gi.height = height;
204 gi.scale = scale;
205 gi.color = col;
206
207 ynGrid* pGrid = new ynGrid;
208 assert(pGrid);
209 pGrid->Init(pDev, &gi);
210

212
213 return YN_OK;
