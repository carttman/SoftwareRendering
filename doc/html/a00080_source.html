1051//! \param iPos 좌표 입력. 3성분.

1053//! \return 성공시 OK, 실패시 FALSE
1054//
1055int B3YenaGraphicsEngine9::_Viewport(
1056 B3YXVECTOR3 iPos[3],
1057 B3YXVECTOR2 oPos[3]
1058 )
1059{
1060 #define vDevSpace iPos[i] //가독성 향상을 위한 메크로 선언.
1061 #define vScreen oPos[i]
1062
1063 float w = (float)m_PresentParam.Width;

1065
1066 for (UINT i = 0; i < m_TopologyVtxSize; i++) //기하도형별 정점수 만큼 처리.
