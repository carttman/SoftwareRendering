175/////////////////////////////////////////////////////////////////////////////
176//
177//! 오브젝트 : 생성 및 필요 리소스 로딩.
178//!
179//! \return 성공시 OK, 실패시 FALSE
180//
181int ObjLoad()
182{
183
184 //기하 (3D 모델) 데이터 : 외부 저장기기(HDD) 에서 로딩되었다 가정합니다.
185 COLVTX Vertices[] = {
186
187 // 정점 좌표는 모델 공간(Model Space, Local Space) 좌표.
188 // 3D 좌표 (x, y, z) 색상( a, r, g, b) a, 곧 Alpha 는 기본값 255 (1.0f)
189 { -0.5f, 0.0f, 0.0f, 0xffff0000 }, //v0, Red.
190 { 0.0f, 1.0f, 0.0f, 0xff00ff00 }, //v1, Green
191 { 0.5f, 0.0f, 0.0f, 0xff00ffff }, //v2, Light-Blue
192
193 /*//Face 1 : 역삼각형.(CCW)
194 { -0.5f, 0.0f, 0.0f, 0xffff0000 }, //v3, Red.
195 { 0.0f,-1.0f, 0.0f, 0xff00ff00 }, //v4, Green
196 { 0.5f, 0.0f, 0.0f, 0xff00ffff }, //v5, Light-Blue
197 */
198 };
199
200
201 // 정점 버퍼 생성.
202 if (YN_FAILED(g_pDevice->CreateVertexBuffer(
203 sizeof(Vertices), // 정점 버퍼 크기 (바이트)
204 0, // 버퍼 사용방식 (기본 0)
205 FVF_COLVTX, // 정점 규격.
206 B3YPOOL_SYSTEMMEM, // 시스템 메모리 사용. (현재는 이 옵션만 유효)
207 &g_pVB, // 성공시 리턴되는 버퍼 운용 객체 인터페이스
208 NULL // 예약됨. NULL.
209 )))
210 {
211 //... 에러 처리...
212 return YN_FALSE;
213 }
214
215 // 버퍼 주소 얻기
216 // 실제 DX 의 그것과는 역활이 다르지만, 동작의 결과는 동일합니다.
217 COLVTX* pBuff = nullptr;
218 if (YN_FAILED(g_pVB->Lock(0, sizeof(Vertices), (void**)&pBuff, 0)))
219 {
220 return YN_FALSE;
221 }
222
223 //버퍼에 정점 데이터 채우기.
224 //<DX> CPU -> VB(GPU/VRAM) 정점 데이터 복사
225 //<Yena> CPU -> VB(SYSMEM) 정점 데이터 복사.
226 memcpy(pBuff, Vertices, sizeof(Vertices));
227
228 //<DX> 작업 완료, 버퍼 사용이 끝나면 버퍼 잠금을 해제(Unlock) 해야합니다.
229 //<Yena> 우리는 이 동작을 간략하게 처리하겠습니다.
230 g_pVB->Unlock();
231
232
233 //... 버퍼 생성 완료...
234 //... 이제 적절한 때에 VB 를 장치에 등록(Binding) 하고 렌더링을 수행하면 되겠습니다.
235 //... 생성된 VB 는 사용자가 제거해야 합니다. (Release 호출)
236
237 return YN_OK;
238}

259//! 오브젝트 : 상태 갱신

262//! \todo2 "주인공" 변환 행렬 결합 연습 (SRT, STR)

264void ObjUpdate(float dTime)
265{
266
267 // 스케일 처리
268 //
269 B3YXMATRIX mScale;
270 B3YXMatrixScaling(&mScale, 1, 1, 1); //원본 100%
271 //B3YXMatrixScaling(&mScale, 2, 1, 1); //원본 200% (x축)
272 //B3YXMatrixScaling(&mScale, 3, 3, 3); //원본 300%
273
274
275
276 //회전 처리 <1>
277 static float angle = 0.0f;
278 //angle += B3Y_PI * dTime; //180 (도)/sec 즉, 1회전에 2초 소요.
279 angle += B3YXToRadian(180) * dTime; //위와 동일코드. 매크로 버전.
280 //angle = ynToRadian(0); //수동 입력.
281
282 B3YXMATRIX mRot; //회선성분이 들어간 행렬 만들기
283 //B3YXMatrixRotationZ(&mRot, angle); //Z 축 기준 회전: mR = Z축회전(angle)
284 //B3YXMatrixRotationX(&mRot, angle); //X 축 기준 회전
285 B3YXMatrixRotationY(&mRot, angle); //Y 축 기준 회전
286 //B3YXMatrixRotationZ(&mRot, B3YXToRadian(0));
287 //B3YXMatrixRotationZ(&mRot, B3YXToRadian(-45));
288
289
290 // 이동 처리 ★
291 //
292 static float mov = 5.0f;
293 if (IsKeyDown(VK_LEFT)) g_vPos.x -= mov * dTime;
294 if (IsKeyDown(VK_RIGHT)) g_vPos.x += mov * dTime;
295 if (IsKeyDown(VK_UP)) g_vPos.z += mov * dTime;
296 if (IsKeyDown(VK_DOWN)) g_vPos.z -= mov * dTime;
297
298 // 이동행렬 만들기.
299 B3YXMATRIX mTrans;
300 B3YXMatrixTranslation(&mTrans, g_vPos.x, g_vPos.y, g_vPos.z); //★
301 //B3YXMatrixTranslation(&mTrans, 3, 0, 0); //이동 테스트1
302 //B3YXMatrixTranslation(&mTrans, 3, 0, 4); //이동 테스트2
303
304 //----------------------------
305 // 최종 변환 행렬 결합.
306 //----------------------------
307 // 렌더링전에 Device 에 설정되어야 합니다.
308 //
309 g_mTM = mScale * mRot * mTrans; //행렬 결합 방향 주의!
310 //g_mTM = mRot; //[테스트]변화시킬 단일 정보만 설정하는 것도 가능.
311
312 /*
313 //DXUT 함수를 통한 행렬 결합.
314 //상기 결과와 동일. 인자의 순서가 곧 결합 방향 .
315 B3YXMatrixMultiply(&g_mTM, &mScale, &mRot); //mTM = mScale * mRot
316 B3YXMatrixMultiply(&g_mTM, &g_mTM, &mTrans); //mTM = mTM * mTrans
317 */
318
319 //----------------------------
320 // 행렬 결합 역회전 테스트.

362///////////////////////////////////////////////////////////////////////////////
363//
364//! 카메라 갱신
365//! 사용자 지정 카메라 정보를 통해 뷰(View) 및 투영(Projection) 변환 행렬을 생성하고
366//! 장치에 설정합니다..
367//!
368//! \see B3YXMatrixLookAtLH, B3YXMatrixPerspectiveFovLH
369//! \see [D3DXMatrixLookAtLH](https://learn.microsoft.com/en-us/windows/win32/direct3d10/d3d10-d3dxmatrixlookatlh), [D3DXMatrixPerspectiveFovLH](https://learn.microsoft.com/en-us/windows/win32/direct3d9/d3dxmatrixperspectivefovlh)
370//! \see [XMMatrixLookAtLH](https://learn.microsoft.com/en-us/windows/win32/api/directxmath/nf-directxmath-xmmatrixlookatlh), [XMMatrixPerspectiveFovLH](https://learn.microsoft.com/en-us/windows/win32/api/directxmath/nf-directxmath-xmmatrixperspectivefovlh)
371//
372void CameraUpdate(float dTime)
373{
374
375 // 뷰 변환 View Transform : 카메라 Body 설정.
376 //..
377 //B3YXVECTOR3 eye(0.0f, 0.0f, -5.0f); //카메라 위치.1
378 B3YXVECTOR3 eye(0.0f, 5.0f, -10.0f); //카메라 위치.2
379 //B3YXVECTOR3 eye(3.0f, 5.0f, -10.0f); //카메라 위치.3 테스트.
380 B3YXVECTOR3 lookat(0.0f, 0.0f, 0.0f); //바라보는 곳.위치.
381 B3YXVECTOR3 up(0.0f, 1.0f, 0.0f); //카메라 상방 벡터.
382 B3YXMatrixLookAtLH(&g_mView, &eye, &lookat, &up);
383
384 //Rendering Device 에 뷰 변환 행렬 설정.
385 g_pDevice->SetTransform(B3YTS_VIEW, &g_mView);
386
387
388 // 투영 변환 Projection Transform : 렌즈 설정.
389 //
390 FLOAT fov = B3YXToRadian(45); //시야각
391 FLOAT aspect = g_Mode.Width / (float)g_Mode.Height; //가로:세로 비율
392 FLOAT zn = 1.0f; //근평면 거리(시야 최소한계)
393 FLOAT zf = 100.0f; //원평면 거리(시야 최대한계)
394 B3YXMatrixPerspectiveFovLH(&g_mProj, fov, aspect, zn, zf);
395
396 //Rendering Device 에 투영 행렬 설정.
397 g_pDevice->SetTransform(B3YTS_PROJECTION, &g_mProj);
398
399}
