161//! \class ynException
162//! \brief 예외처리 헬퍼 클래스
163//! \details COM / DX 예외처리용, (간단버전)(25.0122)
164//!
165//! \see [참고] COM Error Code : https://learn.microsoft.com/en-us/windows/win32/com/com-error-codes-10
166//! \ingroup Yena-Helpers
167//!
168class ynException : public std::exception
169{
170private:
171 HRESULT m_hr;
172 TCHAR m_strhr[1024] = _T("");
173 TCHAR m_hrMsg[2048] = _T("");
174public:
175 ynException () = delete;
176 ynException (HRESULT hr) : m_hr(hr) { getHRMessage(); }
177 virtual ~ynException () = default;
178
179 _NODISCARD HRESULT hr () noexcept { return m_hr; }
180 _NODISCARD virtual const TCHAR* what() noexcept { return m_hrMsg; }
181 // virtual const TCHAR* what() const noexcept override
182
183 const TCHAR* getHRMessage() noexcept
184 {
185 FormatMessage(FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_MAX_WIDTH_MASK, //No-line breaker.
186 0, m_hr, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
187 m_strhr, 1024, NULL);
188 _stprintf_s(m_hrMsg, _T("\n> [Yena][ynException] Failure with HRESULT of 0x%08X : %s"),
189 static_cast<unsigned int>(m_hr), m_strhr);
190 return m_hrMsg;
191 }
192
193 //virtual int print() noexcept{}
194};
